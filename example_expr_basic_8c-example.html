#include <stdio.h>
#include <stdlib.h>
#include <sagui.h>
/* NOTE: Error checking has been omitted to make it clear. */
int main(int argc, const char *argv[]) {
const char *s;
if (argc != 2) {
printf("%s <EXPRESSION>\n", argv[0]);
return EXIT_FAILURE;
}
s = argv[1];
printf("Result: %f\n", sg_expr_calc(s, strlen(s)));
return EXIT_SUCCESS;
}
